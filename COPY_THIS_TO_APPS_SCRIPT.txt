const SPREADSHEET_ID = "1d4LbzEe8091EJ9x6bdas9bnLqZh6nE2qpBHFYarZWMw";
const SHEET_NAME = "Responses";

function setupSheet() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sh = ss.getSheetByName(SHEET_NAME);

  if (!sh) sh = ss.insertSheet(SHEET_NAME);
  sh.clear();

  sh.appendRow([
    "Дата",
    "Имя гостя",
    "Статус",
    "Формат участия",
    "Имя супруга/супруги",
    "Количество персон",
    "Комментарий",
    "Язык",
    "Страница",
    "Устройство"
  ]);

  sh.setFrozenRows(1);

  const header = sh.getRange(1, 1, 1, 10);
  header.setFontWeight("bold");
  header.setBackground("#b8904f");
  header.setFontColor("#ffffff");

  sh.setColumnWidths(1, 10, 180);
  sh.setColumnWidth(7, 300);
}

function doPost(e) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sh = ss.getSheetByName(SHEET_NAME);

  if (!sh) {
    setupSheet();
    sh = ss.getSheetByName(SHEET_NAME);
  }

  const p = e.parameter || {};

  const attendanceText = p.attendance === "yes" ? "Будет" : "Не сможет";

  let companionText = "—";
  if (p.attendance === "yes") {
    companionText = p.companion === "spouse" ? "С супругом/супругой" : "Один/одна";
  }

  sh.appendRow([
    new Date(),
    p.name || "",
    attendanceText,
    companionText,
    p.spouseName || "",
    Number(p.guestsCount || 0),
    p.comment || "",
    p.lang || "",
    p.page || "",
    p.userAgent || ""
  ]);

  return ContentService
    .createTextOutput(JSON.stringify({ ok: true }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sh = ss.getSheetByName(SHEET_NAME);

  let rows = [];

  if (sh && sh.getLastRow() > 1) {
    const values = sh.getDataRange().getDisplayValues();
    const headers = values.shift();

    rows = values.map(row => {
      const item = {};
      headers.forEach((h, i) => item[h] = row[i]);

      return {
        timestamp: item["Дата"] || "",
        name: item["Имя гостя"] || "",
        attendance: item["Статус"] === "Будет" ? "yes" : "no",
        companion: item["Формат участия"] === "С супругом/супругой" ? "spouse" : "alone",
        spouseName: item["Имя супруга/супруги"] || "",
        guestsCount: item["Количество персон"] || 0,
        comment: item["Комментарий"] || "",
        lang: item["Язык"] || ""
      };
    });
  }

  const payload = JSON.stringify({ rows });
  const callback = e.parameter.callback;

  if (callback) {
    return ContentService
      .createTextOutput(callback + "(" + payload + ")")
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }

  return ContentService
    .createTextOutput(payload)
    .setMimeType(ContentService.MimeType.JSON);
}
